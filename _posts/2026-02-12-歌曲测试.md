<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/aplayer/dist/APlayer.min.css">
<script src="https://cdn.jsdelivr.net/npm/aplayer/dist/APlayer.min.js"></script>

<!-- 播放器容器 -->
<div id="aplayer"></div>

<!-- 初始化播放器 -->
<script>
// 适配 Jekyll baseurl
const baseurl = "{{ site.baseurl }}";

// 初始化 APlayer
const ap = new APlayer({
    container: document.getElementById('aplayer'),
    fixed: false, // 非固定模式（嵌入页面）
    autoplay: false, // 禁止自动播放（浏览器限制）
    theme: '#F56C6C', // 播放器主题色
    loop: 'single', // 单曲循环
    volume: 0.7, // 默认音量
    audio: [
        {
            name: '花妖', // 歌曲名
            artist: '刀郎', // 歌手
            url: baseurl + '/assets/audio/花妖.mp3', // 音频路径（拼接 baseurl）
            cover: baseurl + '/assets/img/posts/2026-02-09-美人测试', // 可选：封面图片路径
            // lrc: baseurl + '/assets/audio/花妖.lrc', // 可选：歌词文件路径
        }
        // 新增歌曲：复制上面的对象，逗号分隔
    ]
});
</script>
